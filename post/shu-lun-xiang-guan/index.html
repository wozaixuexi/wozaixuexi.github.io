//Eratosthenes筛法求欧拉函数 O(nlogn)
#include<iostream>
#include<cstdio>
	using namespace std;
	int phi[50005];
int main()
{
	int n=0;
	scanf("%d",&n); n--;
	for(int i=1;i<=n;i++) phi[i]=i;
	for(int i=2;i<=n;i++)
	{
		if(phi[i]!=i) continue;
		for(int j=i;j<=n;j+=i)
			phi[j]=phi[j]/i*(i-1);
	}
	int ans=0;
	for(int i=2;i<=n;i++) ans+=(phi[i]<<1);
	ans+=(n>0)?3:0;
	printf("%d",ans);
	return 0;
}

#include<iostream>
#include<cstdio>
	using namespace std;
	int cnt=0,phi[50005],pri[50005],u[50005];
int main()
{
	int n=0;
	scanf("%d",&n); n--;
	for(int i=2;i<=n;i++)
	{
		if(u[i]==0)
			u[i]=i,pri[++cnt]=i,phi[i]=i-1;
		for(int j=1;j<=cnt;j++)
		{
			if(pri[j]>u[i]||pri[j]>n/i) break;
			u[pri[j]*i]=pri[j];	
			phi[pri[j]*i]=phi[i]*((i%pri[j])?(pri[j]-1):pri[j]);
		}
	}

	for(int i=2;i<=n;i++) ans+=(phi[i]<<1);
	ans+=(n>0)?3:0;
	printf("%d",ans);
