筛法求质数

2020-05-28

对于单个数，可以用试除法在O(n)O(\sqrt{n})O(n​)的时间复杂度下判断一个数是否质数。

对于单个数，还有更快的MillerRobin算法来判断一个数是否是质数。（但我不会）

对于1到n中的数，可以用筛选法来求其中的质数（如下）。

LuoguP3383【模板】线性筛素数

Eratosthenes筛法

复杂度O(n loglog n)O(n\ loglog\ n)O(n loglog n)，但仍不能通过这一题。

//Eratosthenes筛法
#include<iostream>
#include<cstdio>
	using namespace std;
	const int N=1e8+5;
	int ans[N];
	bool vis[N];
int main()
{
	int n=0,q=0,cnt=0;
	scanf("%d%d",&n,&q);
	for(int i=2;i<=n;i++)
	{
		if(vis[i]) continue;
		ans[++cnt]=i;
 		for(int j=i;j<=n/i;j++)
			vis[i*j]=true;
	}
	for(int i=1;i<=q;i++)
	{
		int k=0;
		scanf("%d",&k);
		printf("%d\n",ans[k]);
	}
	return 0;
}

程序中的这一句：for(int j=i;j<=n/i;j++) vis[i*j]=true;，有些写法可能会写为for(int j=i*i;j<=n;j+=i) vis[j]=1;，这种写法需要注意边界问题，因为当数据较大时，i*i可能会爆int，从而引发RE。

线性筛

pri数组用于保存质数，Mn[i]表示i的最小质因数。

使用线性筛，每一个数只会被它的最小质因子标记一次。

时间复杂度O(n)O(n)O(n)。

//线性筛 
#include<iostream>
#include<cstdio>
#include<cstring>
	using namespace std;
int rd()
{
	int x=0;
	char ch=getchar();
	while(ch<'0'||ch>'9') ch=getchar();
	while(ch>='0'&&ch<='9')
		x=(x<<3)+(x<<1)+ch-48,ch=getchar();
	return x;
}
	int cnt=0,pri[10000005],Mn[100000005];
int main()
{
	int n=rd(),q=rd();
	for(int i=2;i<=n;i++)
	{
		if(!Mn[i]) Mn[i]=i,pri[++cnt]=i;
		for(int j=1;j<=cnt;j++)
		{
			if(pri[j]>Mn[i]||pri[j]>n/i) break;
			Mn[pri[j]*i]=pri[j];
		}
	}
	for(int i=1;i<=q;i++)
		printf("%d\n",pri[rd()]);
	return 0;
}
